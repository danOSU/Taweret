```python class MyModel(BaseModel):

def log_likelihood_elementwise(self, y_exp, y_err, model_params):

# Assuming a normal distribution for error y = self.evaluate(model_params) # If y_exp, y_err, y are numpy arrays of same length return np.exp(-(y - y_exp) **2 / (2 * y_err ** 2)) / np.sqrt(2 * np.pi * y_err ** 2))
```
